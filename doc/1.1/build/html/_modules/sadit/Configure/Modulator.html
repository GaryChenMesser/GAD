[docs]    def stage(self):
        """for one stage"""
        r_start = self.run_para['r_start']
        r_end = self.run_para['r_end']
        for i in xrange(self.srv_num): # for each destination
            gen = self.states[i][self.cs[i]]
            if not gen:
                continue
            mod = Modulator(
                    name=self.desc['name'],
                    start=r_start,
                    profile=( (r_end-r_start,), (1,) ) ,
                    generator=gen,
                    )
            self.mod_list.append(mod)
