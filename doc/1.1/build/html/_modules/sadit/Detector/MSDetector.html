    return sortedSrcIP, sortedFlowSize, sortedTime, sortedDur, sortIdx

[docs]def CombineMSRecord(feaSet):
    res = feaSet[0]
    res.append([0]*len(res[0]))
    srv_id = 0
    for fs in feaSet[1:]:
        srv_id += 1
        for i in xrange(len(res)-1):
            res[i].append(fs[i])
        res[-1] = [srv_id] * len(fs[0])
    return res
