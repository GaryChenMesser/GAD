[docs]def FHash(digit, level):
    # return digit[0] + digit[1] * level[0] + digit[2] * level[1] * level[0]
    value = digit[0]
    for i in xrange(len(digit)-1):
        value += digit[i+1] * reduce(operator.mul, level[0:i+1])
    return value

[docs]def ModelBased(qFeaVec, feaQN):
    QLevelNum = reduce(operator.mul, feaQN)
    P = np.zeros( (QLevelNum, QLevelNum) )
    fl = len(qFeaVec[0])
    mp = np.zeros((QLevelNum, ))
    m_list = GetFeatureHashList(qFeaVec, feaQN)

    for i in xrange(fl-1):
        mp[m_list[i]] += 1
        P[m_list[i], m_list[i+1]] += 1
    mp[m_list[fl-1]] += 1

    P = P * 1.0 / (fl-1)
    mp = mp / fl
    return P, mp



[docs]def ModelFree(qFeaVec, feaQN):
    QLevelNum = reduce(operator.mul, feaQN)
    P = np.zeros( (QLevelNum, ) )
    fl = len(qFeaVec[0])
    m_list = GetFeatureHashList(qFeaVec, feaQN)

    for i in range(fl):
        idx = m_list[i]
        try:
            P[idx] += 1
        except:
            import pdb; pdb.set_trace()
    P = P * 1.0 / fl
    assert(abs( np.sum(np.sum(P, 0)) - 1.0) < 0.01)
    return P

[docs]def GetFlowRate(t, cluster):
    '''Get Estimation of Flow Rate For Each User
    input: 1. arrival time , 2 cluster labels flows
    output: Estimated flow arrival rate for each time
    '''
    win = settings.FLOW_RATE_ESTIMATE_WINDOW
    fr = []
    for i in xrange(len(t)):
        idx = Find(t, t[i] - win)
        if idx == -1: idx = 0
        c = cluster[i]
        fr.append( cluster[idx:i].count(c) )

    return fr
