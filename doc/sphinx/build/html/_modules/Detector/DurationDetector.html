[docs]def ModelBased_Dur(gc, gd, gf, K, ND, NF, gr, NR):
    T = K * ND * NF * NR
    P = np.zeros( (T, T) )
    fl = len(gc)
    mp = np.zeros((T, ))
    for i in range(fl-1):
        m = gc[i] + gd[i] * K + gf[i] * K * ND + gr[i] * K * ND * NF
        mp[m] += 1
        n = gc[i+1] + gd[i+1] * K + gf[i+1] * K * ND + gr[i] * K * ND * NF
        P[m, n] += 1
    P = P * 1.0 / (fl-1)
    return P, mp



[docs]def ModelFree_Dur(gc, gd, gf, K, ND, NF, gr, NR):
    T = K * ND * NF * NR
    P = np.zeros( (T, ) )
    fl = len(gc)
    for i in range(fl):
        idx = gc[i] + gd[i] * K + gf[i] * K * ND + gr[i] * K * ND * NF
        P[idx] += 1
    P = P * 1.0 / fl
    assert(abs( np.sum(np.sum(P, 0)) - 1.0) < 0.01)
    return P
